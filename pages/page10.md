```python {all|4-6|2|7-8|all}
# 🤔想不出来?给你一个提示💡!
for i in range(100,1000):
    # 我们已经知道如何把数字 i 变成百位、十位、个位
    a = i // 100 # 百位
    b = i // 10 % 10 # 十位
    c = i % 10 # 个位
    if a**3 + b**3 + c**3 == i : 
        print(i) # 打印正确答案
```
